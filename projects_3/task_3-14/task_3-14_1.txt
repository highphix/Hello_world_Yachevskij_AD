!/bin/bash

FILE="students.txt"

if [ ! -f "$FILE" ]; then
cat > "$FILE" << EOF
Ivan 78
Maria 92
Oleg 67
Anna 85
EOF
echo "Файл $FILE создан."
fi

echo "=== Имена студентов ==="
awk '{print $1}' "$FILE"

echo "=== Оценки студентов ==="
awk '{print $2}' "$FILE"

echo "=== Номер строки и имя ==="
awk '{print NR, $1}' "$FILE"
