!/bin/bash

FILE="students.txt"

if [ ! -f "$FILE" ]; then
echo "Ошибка: файл $FILE не найден. Сначала запустите task_3-14_students.sh"
exit 1
fi

echo "=== Студенты с оценкой выше 80 ==="
awk '$2 > 80 {print $1, $2}' "$FILE"

echo "=== Студенты с оценкой ниже 70 ==="
awk '$2 < 70 {print $1, $2}' "$FILE"

echo "=== Первая строка файла ==="
awk 'NR == 1 {print $0}' "$FILE"
